def PrivateShopSecondToDHMS(time):
    if time < 60:
        return "%.2fseg." % time

    second = int(time % 60)
    minute = int((time / 60) % 60)
    hour = int((time / 60) / 60) % 24
    day = int(int((time / 60) / 60) / 24)

    text = ""

    if day > 0:
        text += str(day) + "d "
        text += " "

    if hour > 0:
        text += str(hour) + "h "
        text += " "

    if minute > 0:
        text += str(minute) + "m "

    if second > 0:
        text += str(second) + "s "

    return text
